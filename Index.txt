  <script>
    // Настройка цены за день (потом поменяешь на свою)
    const PRICE_PER_DAY = 15; // грн за день, пример

    const daysInput = document.getElementById("days");
    const accountInput = document.getElementById("account");
    const payBtn = document.getElementById("payBtn");
    const errorEl = document.getElementById("error");
    const priceHint = document.getElementById("priceHint");

    function updatePriceHint() {
      const days = parseInt(daysInput.value, 10);
      if (!isNaN(days) && days > 0) {
        const total = days * PRICE_PER_DAY;
        priceHint.textContent = total + " грн · " + days + " дн.";
      } else {
        priceHint.textContent = "";
      }
    }

    daysInput.addEventListener("input", updatePriceHint);

    payBtn.addEventListener("click", () => {
      errorEl.textContent = "";
      const days = parseInt(daysInput.value, 10);
      const account = accountInput.value.trim();

      if (isNaN(days) || days <= 0) {
        errorEl.textContent = "Укажи корректное количество дней.";
        return;
      }
      if (!account) {
        errorEl.textContent = "Укажи имя аккаунта.";
        return;
      }

      const amount = days * PRICE_PER_DAY;

      // Здесь вставляешь реальную ссылку MonoPay / своего бэкенда.
      // Пример: редирект на твой сервер, который создаёт счёт и возвращает платёжную ссылку.
      const params = new URLSearchParams({
        account: account,
        days: String(days),
        amount: String(amount)
      });

      // Вариант A: твой backend, который дальше общается с MonoPay
      const url = "https://pay.ludens.school/create?" + params.toString();

      // Вариант B: если у MonoPay есть готовая ссылка с comment/amount – подставляешь туда
      // const url = "https://your-monopay-link.example?sum=" + amount + "&comment=" +
      //             encodeURIComponent("premium:" + account + ":" + days);

      window.location.href = url;
    });
  </script>
